// Les méthodes des tableaux - Push : ajouter un nouvel élément à la fin du tableau const fruits = ["Kiwi", "Apple", "Pineapple"]; fruits.push("Banana"); console.log(fruits); => ["Kiwi", "Apple", "Pineapple", "Banana"] - Unshift : ajouter un nouvel élément au début du tableau const fruits = ["Kiwi", "Apple", "Pineapple"]; fruits.unshift("Strawberry"); console.log(fruits); => ["Strawberry,"Kiwi", "Apple", "Pineapple"] - Pop : supprimer le dernir élément du tableau fruits.pop(); console.log(fruits); => ["Kiwi", "Apple"] - Shift : supprimer le premier élément du tableau fruits.shift(); console.log(fruits); ["Kiwi"]
